def allocate_capital(portfolio, deals):
    # Rank deals by net profit ÷ capital ÷ days held
    deals.sort(key=lambda d: d['score'], reverse=True)
    for deal in deals:
        if portfolio['available_cash'] >= deal['cash_required']:
            portfolio['invested'].append(deal)
            portfolio['available_cash'] -= deal['cash_required']
